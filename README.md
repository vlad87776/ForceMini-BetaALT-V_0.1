import numpy as np
from src.model_core import ForceMiniModel
from src.train_engine import ForceMiniTrainer
from src.data_engine import ForceMiniTokenizer, DataProcessor

# 1. –ü–æ–¥–≥–æ—Ç–æ–≤–∫–∞ –¥–∞–Ω–Ω—ã—Ö
texts = ["–ø–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω—ã–π —Ç–µ–∫—Å—Ç", "–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–π —Ç–µ–∫—Å—Ç", "–Ω–µ–π—Ç—Ä–∞–ª—å–Ω—ã–π —Ç–µ–∫—Å—Ç"]

# 4. –û–±—É—á–µ–Ω–∏–µ –º–æ–¥–µ–ª–∏

    input_size=len(tokenizer.word2idx),
    hidden_size=64,
    output_size=3

# 5. –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ

# –ó–∞–≥—Ä—É–∑–∫–∞ –º–æ–¥–µ–ª–∏

test_text = "—ç—Ç–æ —Ö–æ—Ä–æ—à–∏–π –¥–µ–Ω—å"
test_seq = tokenizer.text_to_sequence(test_text)
test_bow = DataProcessor.create_bow_vector(test_seq, len(tokenizer.word2idx))

prediction = loaded_model.forward(test_bow.reshape(1, -1))
predicted_class = np.argmax(prediction)

print(f"–ö–ª–∞—Å—Å: {predicted_class}, –í–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–∏: {prediction}")
